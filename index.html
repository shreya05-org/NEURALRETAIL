DAX measures reference — key calculations
Total Revenue = SUMX(Sales, [Quantity]*[Price])
Net Revenue = [Gross Rev] - [Returns Loss]
AOV = DIVIDE([Revenue],[Orders])
Repeat Rate = DIVIDE([Repeat Cust],[Total Cust])
YoY Growth = DIVIDE([Rev TY]-[Rev LY],[Rev LY])
CLV = [AOV]*[Avg Freq]*[Avg Lifespan]
